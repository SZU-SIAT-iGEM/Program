### 二、 核心动力学采样 (基于 OpenMM)

该模块负责对输入的16个 PDB 候选结构进行隐式溶剂动力学采样，生成构象演变轨迹。

**脚本文件：`1_run_simulation.py**`

```python
import sys
from openmm.app import *
from openmm import *
from openmm.unit import *

def run_implicit_md(input_pdb, output_dcd, steps=500000):
    print(f"开始处理结构: {input_pdb}")
    
    pdb = PDBFile(input_pdb)
    forcefield = ForceField('amber14-all.xml', 'implicit/obc2.xml')
    system = forcefield.createSystem(pdb.topology, nonbondedMethod=NoCutoff, constraints=HBonds)
    
    integrator = LangevinMiddleIntegrator(310*kelvin, 1/picosecond, 0.002*picoseconds)
    platform = Platform.getPlatformByName('CUDA')
    
    simulation = Simulation(pdb.topology, system, integrator, platform)
    simulation.context.setPositions(pdb.positions)
    
    print("正在执行能量最小化...")
    simulation.minimizeEnergy()
    
    # 设置轨迹输出频率
    simulation.reporters.append(DCDReporter(output_dcd, 1000))
    simulation.reporters.append(StateDataReporter(sys.stdout, 50000, step=True, potentialEnergy=True, temperature=True))
    
    print(f"启动动力学模拟，总步数: {steps}...")
    simulation.step(steps)
    print(f"模拟完成，轨迹文件生成: {output_dcd}\n")

if __name__ == '__main__':
    # 示例调用
    run_implicit_md('inputs/L11_Hybrid.pdb', 'outputs/L11_traj.dcd')

```

### 三、 动力学与空间指标联合提取 (基于 MDAnalysis)

该模块读取 MD 轨迹，同步提取系统稳定性、局部柔性、空间位阻及微重力环境下的扫掠体积指标。

**脚本文件：`2_extract_dynamics.py**`

```python
import MDAnalysis as mda
from MDAnalysi
s.analysis import rms, distances, hydrogenbonds
import numpy as np
import pandas as pd

def extract_metrics(pdb_file, dcd_file, linker_sel_str, enzyme_sel_str, scaffold_sel_str):
    print(f"正在分析轨迹: {dcd_file}")
    u = mda.Universe(pdb_file, dcd_file)
    ref = mda.Universe(pdb_file) 
    
    linker = u.select_atoms(linker_sel_str)
    enzyme = u.select_atoms(enzyme_sel_str)
    scaffold = u.select_atoms(scaffold_sel_str)
    
    results = {'Time_ps': [], 'RMSD': [], 'Distance_Clash': [], 'Rg_Sweeping': [], 'H_Bonds': []}
    
    # 1. 均方根偏差 (RMSD)
    R = rms.RMSD(u, ref, select="backbone")
    R.run()
    
    # 2. 局部灵活性 (RMSF)
    calphas = linker.select_atoms('name CA')
    rmsfer = rms.RMSF(calphas).run()
    
    # 3. 内部氢键网络 (评估刚性螺旋稳定性)
    hbonds = hydrogenbonds.HydrogenBondAnalysis(u, select1=linker_sel_str, select2=linker_sel_str)
    hbonds.run()
    
    for i, ts in enumerate(u.trajectory):
        time = ts.time
        
        # 4. 空间位阻 (最小接触距离)
        dist_mat = distances.distance_array(enzyme.positions, scaffold.positions)
        min_dist = np.min(dist_mat)
        
        # 5. 太空扫掠体积 (回转半径 Rg)
        rg = enzyme.radius_of_gyration()
        
        results['Time_ps'].append(time)
        results['RMSD'].append(R.results.rmsd[i, 2]) 
        results['Distance_Clash'].append(min_dist)
        results['Rg_Sweeping'].append(rg)
        results['H_Bonds'].append(hbonds.results.timeseries[i].shape[0] if i < len(hbonds.results.timeseries) else 0)
        
    df = pd.DataFrame(results)
    output_csv = dcd_file.replace('.dcd', '_metrics.csv')
    df.to_csv(output_csv, index=False)
    
    print("[分析简报]")
    print(f"平均 RMSD: {df['RMSD'].mean():.2f} Å")
    print(f"空间碰撞概率 (< 5Å): {(df['Distance_Clash'] < 5.0).mean() * 100:.1f}%")
    print(f"平均扫掠半径 (Rg): {df['Rg_Sweeping'].mean():.2f} Å")
    print(f"连接肽末端 RMSF: {rmsfer.results.rmsf[-5:].mean():.2f} Å")
    print(f"指标数据已保存至: {output_csv}\n")

if __name__ == '__main__':
    extract_metrics('inputs/L11_Hybrid.pdb', 'outputs/L11_traj.dcd', 
                    'resid 101-140', 'resid 141-300', 'resid 1-100')

```

### 四、 太空专属氧化靶点扫描 (基于 FreeSASA)

该模块从第一性原理出发，量化分析序列中易受活性氧（ROS）攻击的原子暴露面积。

**脚本文件：`3_calculate_sasa.py**`

```python
import freesasa

def calculate_rvi(pdb_file):
    print(f"启动 ROS 脆弱指数 (RVI) 扫描: {pdb_file}")
    
    structure = freesasa.Structure(pdb_file)
    result = freesasa.calc(structure)
    
    rvi_score = 0.0
    vulnerable_count = 0
    
    for i in range(structure.nAtoms()):
        res_name = structure.residueName(i).strip()
        atom_name = structure.atomName(i).strip()
        
        if res_name in ['CYS', 'MET'] and atom_name in ['S', 'SG', 'SD']:
            area = result.atomArea(i)
            rvi_score += area
            vulnerable_count += 1
            print(f"检出高危靶点: {res_name}_{structure.residueNumber(i)} [{atom_name}], SASA = {area:.2f} Å²")
            
    print(f"总计检出硫原子数量: {vulnerable_count}")
    print(f"ROS 脆弱指数 (SASA): {rvi_score:.2f} Å²\n")

if __name__ == '__main__':
    calculate_rvi('inputs/L11_Hybrid.pdb')

```

def generate_comparison_plots(metrics_csv_1, metrics_csv_2, label_1, label_2):

    axes[0].set_ylabel('Density', fontsize=12)
    axes[0].legend(prop={'weight': 'bold'})
    
    # 空间位阻分布图 (Distance)
    sns.boxplot(data=[df1['Distance_Clash'], df2['Distance_Clash']], ax=axes[1], width=0.5)
    axes[1].axhline(y=5.0, color='red', linestyle='--', linewidth=2, label='Clash Threshold')
    axes[1].set_xticklabels([label_1, label_2], fontsize=12)
    axes[1].set_title('Minimum Contact Distance', fontsize=14)
    axes[1].set_ylabel('Distance (Å)', fontsize=12)
    axes[1].legend(prop={'weight': 'bold'})
    
    plt.tight_layout()
